    <script>
        // 点击按钮触发 confetti 效果
        document.getElementById('blessBtn').addEventListener('click', function() {
            alert('生日快乐呀～愿你新的一岁每天都像蛋糕一样甜！');
            createConfetti();
        });

        // 生成 confetti 动画
        function createConfetti() {
            for (let i = 0; i < 50; i++) {
                const confetti = document.createElement('div');
                confetti.classList.add('confetti');
                confetti.style.left = Math.random() * 100 + 'vw';
                confetti.style.animation = `fall${Math.floor(Math.random() * 3) + 1} ${2 + Math.random() * 3}s linear forwards`;
                confetti.style.backgroundColor = getRandomColor();
                document.body.appendChild(confetti);

                // 随机颜色
                function getRandomColor() {
                    const colors = ['#ff3e6c', '#ff9966', '#ffcc66', '#99ff99', '#66b3ff', '#cc99ff'];
                    return colors[Math.floor(Math.random() * colors.length)];
                }
            }

            // 添加动画样式
            const style = document.createElement('style');
            style.textContent = `
                @keyframes fall1 {
                    0% { top: -10vh; transform: rotate(0deg); }
                    100% { top: 100vh; transform: rotate(360deg); }
                }
                @keyframes fall2 {
                    0% { top: -10vh; transform: rotate(10deg); }
                    100% { top: 100vh; transform: rotate(720deg); }
                }
                @keyframes fall3 {
                    0% { top: -10vh; transform: rotate(-10deg); }
                    100% { top: 100vh; transform: rotate(-360deg); }
                }
            `;
            document.head.appendChild(style);

            // 3秒后清除 confetti
            setTimeout(() => {
                const confetti = document.querySelectorAll('.confetti');
                confetti.forEach(item => item.remove());
            }, 3000);
        }
    </script>
